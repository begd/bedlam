  8 /** @namespace Base64 encoding/decoding */
  9 sjcl.codec.base64 = {
 10   /** The base64 alphabet.
 11    * @private
 12    */
 13   _chars: "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/",
 14   
 15   /** Convert from a bitArray to a base64 string. */
 16   fromBits: function (arr, _noEquals, _url) {
 17     var out = "", i, bits=0, c = sjcl.codec.base64._chars, ta=0, bl = sjcl.bitArray.bitLength(arr);
 18     if (_url) {
 19       c = c.substr(0,62) + '-_';
 20     }
 21     for (i=0; out.length * 6 < bl; ) {
 22       out += c.charAt((ta ^ arr[i]>>>bits) >>> 26);
 23       if (bits < 6) {
 24         ta = arr[i] << (6-bits);
 25         bits += 26;
 26         i++;
 27       } else {
 28         ta <<= 6;
 29         bits -= 6;
 30       }
 31     }
 32     while ((out.length & 3) && !_noEquals) { out += "="; }
 33     return out;
 34   },
 35   
 36   /** Convert from a base64 string to a bitArray */
 37   toBits: function(str, _url) {
 38     str = str.replace(/\s|=/g,'');
 39     var out = [], i, bits=0, c = sjcl.codec.base64._chars, ta=0, x;
 40     if (_url) {
 41       c = c.substr(0,62) + '-_';
 42     }
 43     for (i=0; i<str.length; i++) {
 44       x = c.indexOf(str.charAt(i));
 45       if (x < 0) {
 46         throw new sjcl.exception.invalid("this isn't base64!");
 47       }
 48       if (bits > 26) {
 49         bits -= 26;
 50         out.push(ta ^ x>>>bits);
 51         ta  = x << (32-bits);
 52       } else {
 53         bits += 6;
 54         ta ^= x << (32-bits);
 55       }
 56     }
 57     if (bits&56) {
 58       out.push(sjcl.bitArray.partial(bits&56, ta, 1));
 59     }
 60     return out;
 61   }
 62 };
 63 
 64 sjcl.codec.base64url = {
 65   fromBits: function (arr) { return sjcl.codec.base64.fromBits(arr,1,1); },
 66   toBits: function (str) { return sjcl.codec.base64.toBits(str,1); }
 67 };
