 16 /**
 17  * Context for a SHA-256 operation in progress.
 18  * @constructor
 19  * @class Secure Hash Algorithm, 256 bits.
 20  */
 21 sjcl.hash.sha256 = function (hash) {
 22   if (!this._key[0]) { this._precompute(); }
 23   if (hash) {
 24     this._h = hash._h.slice(0);
 25     this._buffer = hash._buffer.slice(0);
 26     this._length = hash._length;
 27   } else {
 28     this.reset();
 29   }
 30 };

 32 /**
 33  * Hash a string or an array of words.
 34  * @static
 35  * @param {bitArray|String} data the data to hash.
 36  * @return {bitArray} The hash value, an array of 16 big-endian words.
 37  */
 38 sjcl.hash.sha256.hash = function (data) {
 39   return (new sjcl.hash.sha256()).update(data).finalize();
 40 };

 42 sjcl.hash.sha256.prototype = {
 43   /**
 44    * The hash's block size, in bits.
 45    * @constant
 46    */
 47   blockSize: 512,

 49   /**
 50    * Reset the hash state.
 51    * @return this
 52    */
 53   reset:function () {
 54     this._h = this._init.slice(0);
 55     this._buffer = [];
 56     this._length = 0;
 57     return this;
 58   },
 59   
 60   /**
 61    * Input several words to the hash.
 62    * @param {bitArray|String} data the data to hash.
 63    * @return this
 64    */
 65   update: function (data) {
 66     if (typeof data === "string") {
 67       data = sjcl.codec.utf8String.toBits(data);
 68     }
 69     var i, b = this._buffer = sjcl.bitArray.concat(this._buffer, data),
 70         ol = this._length,
 71         nl = this._length = ol + sjcl.bitArray.bitLength(data);
 72     for (i = 512+ol & -512; i <= nl; i+= 512) {
 73       this._block(b.splice(0,16));
 74     }
 75     return this;
 76   },
 77   
 78   /**
 79    * Complete hashing and output the hash value.
 80    * @return {bitArray} The hash value, an array of 8 big-endian words.
 81    */
 82   finalize:function () {
 83     var i, b = this._buffer, h = this._h;
 84 
 85     // Round out and push the buffer
 86     b = sjcl.bitArray.concat(b, [sjcl.bitArray.partial(1,1)]);
 87     
 88     // Round out the buffer to a multiple of 16 words, less the 2 length words.
 89     for (i = b.length + 2; i & 15; i++) {
 90       b.push(0);
 91     }
 92     
 93     // append the length
 94     b.push(Math.floor(this._length / 0x100000000));
 95     b.push(this._length | 0);
 96 
 97     while (b.length) {
 98       this._block(b.splice(0,16));
 99     }
100 
101     this.reset();
102     return h;
103   },

132   /**
133    * Function to precompute _init and _key.
134    * @private
135    */
136   _precompute: function () {
137     var i = 0, prime = 2, factor;
138 
139     function frac(x) { return (x-Math.floor(x)) * 0x100000000 | 0; }
140 
141     outer: for (; i<64; prime++) {
142       for (factor=2; factor*factor <= prime; factor++) {
143         if (prime % factor === 0) {
144           // not a prime
145           continue outer;
146         }
147       }
148       
149       if (i<8) {
150         this._init[i] = frac(Math.pow(prime, 1/2));
151       }
152       this._key[i] = frac(Math.pow(prime, 1/3));
153       i++;
154     }
155   },
156   
157   /**
158    * Perform one cycle of SHA-256.
159    * @param {bitArray} words one block of words.
160    * @private
161    */
162   _block:function (words) {  
163     var i, tmp, a, b,
164       w = words.slice(0),
165       h = this._h,
166       k = this._key,
167       h0 = h[0], h1 = h[1], h2 = h[2], h3 = h[3],
168       h4 = h[4], h5 = h[5], h6 = h[6], h7 = h[7];
169 
170     /* Rationale for placement of |0 :
171      * If a value can overflow is original 32 bits by a factor of more than a few
172      * million (2^23 ish), there is a possibility that it might overflow the
173      * 53-bit mantissa and lose precision.
174      *
175      * To avoid this, we clamp back to 32 bits by |'ing with 0 on any value that
176      * propagates around the loop, and on the hash state h[].  I don't believe
177      * that the clamps on h4 and on h0 are strictly necessary, but it's close
178      * (for h4 anyway), and better safe than sorry.
179      *
180      * The clamps on h[] are necessary for the output to be correct even in the
181      * common case and for short inputs.
182      */
183     for (i=0; i<64; i++) {
184       // load up the input word for this round
185       if (i<16) {
186         tmp = w[i];
187       } else {
188         a   = w[(i+1 ) & 15];
189         b   = w[(i+14) & 15];
190         tmp = w[i&15] = ((a>>>7  ^ a>>>18 ^ a>>>3  ^ a<<25 ^ a<<14) + 
191                          (b>>>17 ^ b>>>19 ^ b>>>10 ^ b<<15 ^ b<<13) +
192                          w[i&15] + w[(i+9) & 15]) | 0;
193       }
194       
195       tmp = (tmp + h7 + (h4>>>6 ^ h4>>>11 ^ h4>>>25 ^ h4<<26 ^ h4<<21 ^ h4<<7) +  (h6 ^ h4&(h5^h6)) + k[i]); // | 0;
196       
197       // shift register
198       h7 = h6; h6 = h5; h5 = h4;
199       h4 = h3 + tmp | 0;
200       h3 = h2; h2 = h1; h1 = h0;
201 
202       h0 = (tmp +  ((h1&h2) ^ (h3&(h1^h2))) + (h1>>>2 ^ h1>>>13 ^ h1>>>22 ^ h1<<30 ^ h1<<19 ^ h1<<10)) | 0;
203     }
204 
205     h[0] = h[0]+h0 | 0;
206     h[1] = h[1]+h1 | 0;
207     h[2] = h[2]+h2 | 0;
208     h[3] = h[3]+h3 | 0;
209     h[4] = h[4]+h4 | 0;
210     h[5] = h[5]+h5 | 0;
211     h[6] = h[6]+h6 | 0;
212     h[7] = h[7]+h7 | 0;
213   }
214 };
