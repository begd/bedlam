  8 /** @namespace Arrays of bits, encoded as arrays of Numbers.
  9  *
 10  * @description
 11  * <p>
 12  * These objects are the currency accepted by SJCL's crypto functions.
 13  * </p>
 14  *
 15  * <p>
 16  * Most of our crypto primitives operate on arrays of 4-byte words internally,
 17  * but many of them can take arguments that are not a multiple of 4 bytes.
 18  * This library encodes arrays of bits (whose size need not be a multiple of 8
 19  * bits) as arrays of 32-bit words.  The bits are packed, big-endian, into an
 20  * array of words, 32 bits at a time.  Since the words are double-precision
 21  * floating point numbers, they fit some extra data.  We use this (in a private,
 22  * possibly-changing manner) to encode the number of bits actually  present
 23  * in the last word of the array.
 24  * </p>
 25  *
 26  * <p>
 27  * Because bitwise ops clear this out-of-band data, these arrays can be passed
 28  * to ciphers like AES which want arrays of words.
 29  * </p>
 30  */
 31 sjcl.bitArray = {
 32   /**
 33    * Array slices in units of bits.
 34    * @param {bitArray} a The array to slice.
 35    * @param {Number} bstart The offset to the start of the slice, in bits.
 36    * @param {Number} bend The offset to the end of the slice, in bits.  If this is undefined,
 37    * slice until the end of the array.
 38    * @return {bitArray} The requested slice.
 39    */
 40   bitSlice: function (a, bstart, bend) {
 41     a = sjcl.bitArray._shiftRight(a.slice(bstart/32), 32 - (bstart & 31)).slice(1);
 42     return (bend === undefined) ? a : sjcl.bitArray.clamp(a, bend-bstart);
 43   },
 44 
 45   /**
 46    * Extract a number packed into a bit array.
 47    * @param {bitArray} a The array to slice.
 48    * @param {Number} bstart The offset to the start of the slice, in bits.
 49    * @param {Number} length The length of the number to extract.
 50    * @return {Number} The requested slice.
 51    */
 52   extract: function(a, bstart, blength) {
 53     // FIXME: this Math.floor is not necessary at all, but for some reason
 54     // seems to suppress a bug in the Chromium JIT.
 55     var x, sh = Math.floor((-bstart-blength) & 31);
 56     if ((bstart + blength - 1 ^ bstart) & -32) {
 57       // it crosses a boundary
 58       x = (a[bstart/32|0] << (32 - sh)) ^ (a[bstart/32+1|0] >>> sh);
 59     } else {
 60       // within a single word
 61       x = a[bstart/32|0] >>> sh;
 62     }
 63     return x & ((1<<blength) - 1);
 64   },
 65 
 66   /**
 67    * Concatenate two bit arrays.
 68    * @param {bitArray} a1 The first array.
 69    * @param {bitArray} a2 The second array.
 70    * @return {bitArray} The concatenation of a1 and a2.
 71    */
 72   concat: function (a1, a2) {
 73     if (a1.length === 0 || a2.length === 0) {
 74       return a1.concat(a2);
 75     }
 76     
 77     var out, i, last = a1[a1.length-1], shift = sjcl.bitArray.getPartial(last);
 78     if (shift === 32) {
 79       return a1.concat(a2);
 80     } else {
 81       return sjcl.bitArray._shiftRight(a2, shift, last|0, a1.slice(0,a1.length-1));
 82     }
 83   },
 84 
 85   /**
 86    * Find the length of an array of bits.
 87    * @param {bitArray} a The array.
 88    * @return {Number} The length of a, in bits.
 89    */
 90   bitLength: function (a) {
 91     var l = a.length, x;
 92     if (l === 0) { return 0; }
 93     x = a[l - 1];
 94     return (l-1) * 32 + sjcl.bitArray.getPartial(x);
 95   },
 96 
 97   /**
 98    * Truncate an array.
 99    * @param {bitArray} a The array.
100    * @param {Number} len The length to truncate to, in bits.
101    * @return {bitArray} A new array, truncated to len bits.
102    */
103   clamp: function (a, len) {
104     if (a.length * 32 < len) { return a; }
105     a = a.slice(0, Math.ceil(len / 32));
106     var l = a.length;
107     len = len & 31;
108     if (l > 0 && len) {
109       a[l-1] = sjcl.bitArray.partial(len, a[l-1] & 0x80000000 >> (len-1), 1);
110     }
111     return a;
112   },
113 
114   /**
115    * Make a partial word for a bit array.
116    * @param {Number} len The number of bits in the word.
117    * @param {Number} x The bits.
118    * @param {Number} [0] _end Pass 1 if x has already been shifted to the high side.
119    * @return {Number} The partial word.
120    */
121   partial: function (len, x, _end) {
122     if (len === 32) { return x; }
123     return (_end ? x|0 : x << (32-len)) + len * 0x10000000000;
124   },
125 
126   /**
127    * Get the number of bits used by a partial word.
128    * @param {Number} x The partial word.
129    * @return {Number} The number of bits used by the partial word.
130    */
131   getPartial: function (x) {
132     return Math.round(x/0x10000000000) || 32;
133   },
134 
135   /**
136    * Compare two arrays for equality in a predictable amount of time.
137    * @param {bitArray} a The first array.
138    * @param {bitArray} b The second array.
139    * @return {boolean} true if a == b; false otherwise.
140    */
141   equal: function (a, b) {
142     if (sjcl.bitArray.bitLength(a) !== sjcl.bitArray.bitLength(b)) {
143       return false;
144     }
145     var x = 0, i;
146     for (i=0; i<a.length; i++) {
147       x |= a[i]^b[i];
148     }
149     return (x === 0);
150   },
151 
152   /** Shift an array right.
153    * @param {bitArray} a The array to shift.
154    * @param {Number} shift The number of bits to shift.
155    * @param {Number} [carry=0] A byte to carry in
156    * @param {bitArray} [out=[]] An array to prepend to the output.
157    * @private
158    */
159   _shiftRight: function (a, shift, carry, out) {
160     var i, last2=0, shift2;
161     if (out === undefined) { out = []; }
162     
163     for (; shift >= 32; shift -= 32) {
164       out.push(carry);
165       carry = 0;
166     }
167     if (shift === 0) {
168       return out.concat(a);
169     }
170     
171     for (i=0; i<a.length; i++) {
172       out.push(carry | a[i]>>>shift);
173       carry = a[i] << (32-shift);
174     }
175     last2 = a.length ? a[a.length-1] : 0;
176     shift2 = sjcl.bitArray.getPartial(last2);
177     out.push(sjcl.bitArray.partial(shift+shift2 & 31, (shift + shift2 > 32) ? carry : out.pop(),1));
178     return out;
179   },
180   
181   /** xor a block of 4 words together.
182    * @private
183    */
184   _xor4: function(x,y) {
185     return [x[0]^y[0],x[1]^y[1],x[2]^y[2],x[3]^y[3]];
186   }
187 };
