 18 /**
 19  * Schedule out an AES key for both encryption and decryption.  This
 20  * is a low-level class.  Use a cipher mode to do bulk encryption.
 21  *
 22  * @constructor
 23  * @param {Array} key The key as an array of 4, 6 or 8 words.
 24  *
 25  * @class Advanced Encryption Standard (low-level interface)
 26  */
 27 sjcl.cipher.aes = function (key) {
 28   if (!this._tables[0][0][0]) {
 29     this._precompute();
 30   }
 31   
 32   var i, j, tmp,
 33     encKey, decKey,
 34     sbox = this._tables[0][4], decTable = this._tables[1],
 35     keyLen = key.length, rcon = 1;
 36   
 37   if (keyLen !== 4 && keyLen !== 6 && keyLen !== 8) {
 38     throw new sjcl.exception.invalid("invalid aes key size");
 39   }
 40   
 41   this._key = [encKey = key.slice(0), decKey = []];
 42   
 43   // schedule encryption keys
 44   for (i = keyLen; i < 4 * keyLen + 28; i++) {
 45     tmp = encKey[i-1];
 46     
 47     // apply sbox
 48     if (i%keyLen === 0 || (keyLen === 8 && i%keyLen === 4)) {
 49       tmp = sbox[tmp>>>24]<<24 ^ sbox[tmp>>16&255]<<16 ^ sbox[tmp>>8&255]<<8 ^ sbox[tmp&255];
 50       
 51       // shift rows and add rcon
 52       if (i%keyLen === 0) {
 53         tmp = tmp<<8 ^ tmp>>>24 ^ rcon<<24;
 54         rcon = rcon<<1 ^ (rcon>>7)*283;
 55       }
 56     }
 57     
 58     encKey[i] = encKey[i-keyLen] ^ tmp;
 59   }
 60   
 61   // schedule decryption keys
 62   for (j = 0; i; j++, i--) {
 63     tmp = encKey[j&3 ? i : i - 4];
 64     if (i<=4 || j<4) {
 65       decKey[j] = tmp;
 66     } else {
 67       decKey[j] = decTable[0][sbox[tmp>>>24      ]] ^
 68                   decTable[1][sbox[tmp>>16  & 255]] ^
 69                   decTable[2][sbox[tmp>>8   & 255]] ^
 70                   decTable[3][sbox[tmp      & 255]];
 71     }
 72   }
 73 };
 74 
 75 sjcl.cipher.aes.prototype = {
 76   // public
 77   /* Something like this might appear here eventually
 78   name: "AES",
 79   blockSize: 4,
 80   keySizes: [4,6,8],
 81   */
 82   
 83   /**
 84    * Encrypt an array of 4 big-endian words.
 85    * @param {Array} data The plaintext.
 86    * @return {Array} The ciphertext.
 87    */
 88   encrypt:function (data) { return this._crypt(data,0); },
 89   
 90   /**
 91    * Decrypt an array of 4 big-endian words.
 92    * @param {Array} data The ciphertext.
 93    * @return {Array} The plaintext.
 94    */
 95   decrypt:function (data) { return this._crypt(data,1); },
 96   
 97   /**
 98    * The expanded S-box and inverse S-box tables.  These will be computed
 99    * on the client so that we don't have to send them down the wire.
100    *
101    * There are two tables, _tables[0] is for encryption and
102    * _tables[1] is for decryption.
103    *
104    * The first 4 sub-tables are the expanded S-box with MixColumns.  The
105    * last (_tables[01][4]) is the S-box itself.
106    *
107    * @private
108    */
109   _tables: [[[],[],[],[],[]],[[],[],[],[],[]]],
110 
111   /**
112    * Expand the S-box tables.
113    *
114    * @private
115    */
116   _precompute: function () {
117    var encTable = this._tables[0], decTable = this._tables[1],
118        sbox = encTable[4], sboxInv = decTable[4],
119        i, x, xInv, d=[], th=[], x2, x4, x8, s, tEnc, tDec;
120 
121     // Compute double and third tables
122    for (i = 0; i < 256; i++) {
123      th[( d[i] = i<<1 ^ (i>>7)*283 )^i]=i;
124    }
125    
126    for (x = xInv = 0; !sbox[x]; x ^= x2 || 1, xInv = th[xInv] || 1) {
127      // Compute sbox
128      s = xInv ^ xInv<<1 ^ xInv<<2 ^ xInv<<3 ^ xInv<<4;
129      s = s>>8 ^ s&255 ^ 99;
130      sbox[x] = s;
131      sboxInv[s] = x;
132      
133      // Compute MixColumns
134      x8 = d[x4 = d[x2 = d[x]]];
135      tDec = x8*0x1010101 ^ x4*0x10001 ^ x2*0x101 ^ x*0x1010100;
136      tEnc = d[s]*0x101 ^ s*0x1010100;
137      
138      for (i = 0; i < 4; i++) {
139        encTable[i][x] = tEnc = tEnc<<24 ^ tEnc>>>8;
140        decTable[i][s] = tDec = tDec<<24 ^ tDec>>>8;
141      }
142    }
143    
144    // Compactify.  Considerable speedup on Firefox.
145    for (i = 0; i < 5; i++) {
146      encTable[i] = encTable[i].slice(0);
147      decTable[i] = decTable[i].slice(0);
148    }
149   },
150   
151   /**
152    * Encryption and decryption core.
153    * @param {Array} input Four words to be encrypted or decrypted.
154    * @param dir The direction, 0 for encrypt and 1 for decrypt.
155    * @return {Array} The four encrypted or decrypted words.
156    * @private
157    */
158   _crypt:function (input, dir) {
159     if (input.length !== 4) {
160       throw new sjcl.exception.invalid("invalid aes block size");
161     }
162     
163     var key = this._key[dir],
164         // state variables a,b,c,d are loaded with pre-whitened data
165         a = input[0]           ^ key[0],
166         b = input[dir ? 3 : 1] ^ key[1],
167         c = input[2]           ^ key[2],
168         d = input[dir ? 1 : 3] ^ key[3],
169         a2, b2, c2,
170         
171         nInnerRounds = key.length/4 - 2,
172         i,
173         kIndex = 4,
174         out = [0,0,0,0],
175         table = this._tables[dir],
176         
177         // load up the tables
178         t0    = table[0],
179         t1    = table[1],
180         t2    = table[2],
181         t3    = table[3],
182         sbox  = table[4];
183  
184     // Inner rounds.  Cribbed from OpenSSL.
185     for (i = 0; i < nInnerRounds; i++) {
186       a2 = t0[a>>>24] ^ t1[b>>16 & 255] ^ t2[c>>8 & 255] ^ t3[d & 255] ^ key[kIndex];
187       b2 = t0[b>>>24] ^ t1[c>>16 & 255] ^ t2[d>>8 & 255] ^ t3[a & 255] ^ key[kIndex + 1];
188       c2 = t0[c>>>24] ^ t1[d>>16 & 255] ^ t2[a>>8 & 255] ^ t3[b & 255] ^ key[kIndex + 2];
189       d  = t0[d>>>24] ^ t1[a>>16 & 255] ^ t2[b>>8 & 255] ^ t3[c & 255] ^ key[kIndex + 3];
190       kIndex += 4;
191       a=a2; b=b2; c=c2;
192     }
193         
194     // Last round.
195     for (i = 0; i < 4; i++) {
196       out[dir ? 3&-i : i] =
197         sbox[a>>>24      ]<<24 ^ 
198         sbox[b>>16  & 255]<<16 ^
199         sbox[c>>8   & 255]<<8  ^
200         sbox[d      & 255]     ^
201         key[kIndex++];
202       a2=a; a=b; b=c; c=d; d=a2;
203     }
204     
205     return out;
206   }
207 };
