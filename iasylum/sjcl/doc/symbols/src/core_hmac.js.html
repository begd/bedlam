  8 /** HMAC with the specified hash function.
  9  * @constructor
 10  * @param {bitArray} key the key for HMAC.
 11  * @param {Object} [hash=sjcl.hash.sha256] The hash function to use.
 12  */
 13 sjcl.misc.hmac = function (key, Hash) {
 14   this._hash = Hash = Hash || sjcl.hash.sha256;
 15   var exKey = [[],[]], i,
 16       bs = Hash.prototype.blockSize / 32;
 17   this._baseHash = [new Hash(), new Hash()];
 18 
 19   if (key.length > bs) {
 20     key = Hash.hash(key);
 21   }
 22   
 23   for (i=0; i<bs; i++) {
 24     exKey[0][i] = key[i]^0x36363636;
 25     exKey[1][i] = key[i]^0x5C5C5C5C;
 26   }
 27   
 28   this._baseHash[0].update(exKey[0]);
 29   this._baseHash[1].update(exKey[1]);
 30   this._resultHash = new Hash(this._baseHash[0]);
 31 };
 32 
 33 /** HMAC with the specified hash function.  Also called encrypt since it's a prf.
 34  * @param {bitArray|String} data The data to mac.
 35  */
 36 sjcl.misc.hmac.prototype.encrypt = sjcl.misc.hmac.prototype.mac = function (data) {
 37   if (!this._updated) {
 38     this.update(data);
 39     return this.digest(data);
 40   } else {
 41     throw new sjcl.exception.invalid("encrypt on already updated hmac called!");
 42   }
 43 };
 44 
 45 sjcl.misc.hmac.prototype.reset = function () {
 46   this._resultHash = new this._hash(this._baseHash[0]);
 47   this._updated = false;
 48 };
 49 
 50 sjcl.misc.hmac.prototype.update = function (data) {
 51   this._updated = true;
 52   this._resultHash.update(data);
 53 };
 54 
 55 sjcl.misc.hmac.prototype.digest = function () {
 56   var w = this._resultHash.finalize(), result = new (this._hash)(this._baseHash[1]).update(w).finalize();
 57 
 58   this.reset();
 59 
 60   return result;
 61 };
