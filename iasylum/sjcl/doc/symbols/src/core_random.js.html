  9 /** @constructor
 10  * @class Random number generator
 11  * @description
 12  * <b>Use sjcl.random as a singleton for this class!</b>
 13  * <p>
 14  * This random number generator is a derivative of Ferguson and Schneier's
 15  * generator Fortuna.  It collects entropy from various events into several
 16  * pools, implemented by streaming SHA-256 instances.  It differs from
 17  * ordinary Fortuna in a few ways, though.
 18  * </p>
 19  *
 20  * <p>
 21  * Most importantly, it has an entropy estimator.  This is present because
 22  * there is a strong conflict here between making the generator available
 23  * as soon as possible, and making sure that it doesn't "run on empty".
 24  * In Fortuna, there is a saved state file, and the system is likely to have
 25  * time to warm up.
 26  * </p>
 27  *
 28  * <p>
 29  * Second, because users are unlikely to stay on the page for very long,
 30  * and to speed startup time, the number of pools increases logarithmically:
 31  * a new pool is created when the previous one is actually used for a reseed.
 32  * This gives the same asymptotic guarantees as Fortuna, but gives more
 33  * entropy to early reseeds.
 34  * </p>
 35  *
 36  * <p>
 37  * The entire mechanism here feels pretty klunky.  Furthermore, there are
 38  * several improvements that should be made, including support for
 39  * dedicated cryptographic functions that may be present in some browsers;
 40  * state files in local storage; cookies containing randomness; etc.  So
 41  * look for improvements in future versions.
 42  * </p>
 43  */
 44 sjcl.prng = function(defaultParanoia) {
 45   
 46   /* private */
 47   this._pools                   = [new sjcl.hash.sha256()];
 48   this._poolEntropy             = [0];
 49   this._reseedCount             = 0;
 50   this._robins                  = {};
 51   this._eventId                 = 0;
 52   
 53   this._collectorIds            = {};
 54   this._collectorIdNext         = 0;
 55   
 56   this._strength                = 0;
 57   this._poolStrength            = 0;
 58   this._nextReseed              = 0;
 59   this._key                     = [0,0,0,0,0,0,0,0];
 60   this._counter                 = [0,0,0,0];
 61   this._cipher                  = undefined;
 62   this._defaultParanoia         = defaultParanoia;
 63   
 64   /* event listener stuff */
 65   this._collectorsStarted       = false;
 66   this._callbacks               = {progress: {}, seeded: {}};
 67   this._callbackI               = 0;
 68   
 69   /* constants */
 70   this._NOT_READY               = 0;
 71   this._READY                   = 1;
 72   this._REQUIRES_RESEED         = 2;
 73 
 74   this._MAX_WORDS_PER_BURST     = 65536;
 75   this._PARANOIA_LEVELS         = [0,48,64,96,128,192,256,384,512,768,1024];
 76   this._MILLISECONDS_PER_RESEED = 30000;
 77   this._BITS_PER_RESEED         = 80;
 78 };
 79  
 80 sjcl.prng.prototype = {
 81   /** Generate several random words, and return them in an array.
 82    * A word consists of 32 bits (4 bytes)
 83    * @param {Number} nwords The number of words to generate.
 84    */
 85   randomWords: function (nwords, paranoia) {
 86     var out = [], i, readiness = this.isReady(paranoia), g;
 87   
 88     if (readiness === this._NOT_READY) {
 89       throw new sjcl.exception.notReady("generator isn't seeded");
 90     } else if (readiness & this._REQUIRES_RESEED) {
 91       this._reseedFromPools(!(readiness & this._READY));
 92     }
 93   
 94     for (i=0; i<nwords; i+= 4) {
 95       if ((i+1) % this._MAX_WORDS_PER_BURST === 0) {
 96         this._gate();
 97       }
 98    
 99       g = this._gen4words();
100       out.push(g[0],g[1],g[2],g[3]);
101     }
102     this._gate();
103   
104     return out.slice(0,nwords);
105   },
106   
107   setDefaultParanoia: function (paranoia, allowZeroParanoia) {
108     if (paranoia === 0 && allowZeroParanoia !== "Setting paranoia=0 will ruin your security; use it only for testing") {
109       throw "Setting paranoia=0 will ruin your security; use it only for testing";
110     }
111 
112     this._defaultParanoia = paranoia;
113   },
114   
115   /**
116    * Add entropy to the pools.
117    * @param data The entropic value.  Should be a 32-bit integer, array of 32-bit integers, or string
118    * @param {Number} estimatedEntropy The estimated entropy of data, in bits
119    * @param {String} source The source of the entropy, eg "mouse"
120    */
121   addEntropy: function (data, estimatedEntropy, source) {
122     source = source || "user";
123   
124     var id,
125       i, tmp,
126       t = (new Date()).valueOf(),
127       robin = this._robins[source],
128       oldReady = this.isReady(), err = 0, objName;
129       
130     id = this._collectorIds[source];
131     if (id === undefined) { id = this._collectorIds[source] = this._collectorIdNext ++; }
132       
133     if (robin === undefined) { robin = this._robins[source] = 0; }
134     this._robins[source] = ( this._robins[source] + 1 ) % this._pools.length;
135   
136     switch(typeof(data)) {
137       
138     case "number":
139       if (estimatedEntropy === undefined) {
140         estimatedEntropy = 1;
141       }
142       this._pools[robin].update([id,this._eventId++,1,estimatedEntropy,t,1,data|0]);
143       break;
144       
145     case "object":
146       objName = Object.prototype.toString.call(data);
147       if (objName === "[object Uint32Array]") {
148         tmp = [];
149         for (i = 0; i < data.length; i++) {
150           tmp.push(data[i]);
151         }
152         data = tmp;
153       } else {
154         if (objName !== "[object Array]") {
155           err = 1;
156         }
157         for (i=0; i<data.length && !err; i++) {
158           if (typeof(data[i]) !== "number") {
159             err = 1;
160           }
161         }
162       }
163       if (!err) {
164         if (estimatedEntropy === undefined) {
165           /* horrible entropy estimator */
166           estimatedEntropy = 0;
167           for (i=0; i<data.length; i++) {
168             tmp= data[i];
169             while (tmp>0) {
170               estimatedEntropy++;
171               tmp = tmp >>> 1;
172             }
173           }
174         }
175         this._pools[robin].update([id,this._eventId++,2,estimatedEntropy,t,data.length].concat(data));
176       }
177       break;
178       
179     case "string":
180       if (estimatedEntropy === undefined) {
181        /* English text has just over 1 bit per character of entropy.
182         * But this might be HTML or something, and have far less
183         * entropy than English...  Oh well, let's just say one bit.
184         */
185        estimatedEntropy = data.length;
186       }
187       this._pools[robin].update([id,this._eventId++,3,estimatedEntropy,t,data.length]);
188       this._pools[robin].update(data);
189       break;
190       
191     default:
192       err=1;
193     }
194     if (err) {
195       throw new sjcl.exception.bug("random: addEntropy only supports number, array of numbers or string");
196     }
197   
198     /* record the new strength */
199     this._poolEntropy[robin] += estimatedEntropy;
200     this._poolStrength += estimatedEntropy;
201   
202     /* fire off events */
203     if (oldReady === this._NOT_READY) {
204       if (this.isReady() !== this._NOT_READY) {
205         this._fireEvent("seeded", Math.max(this._strength, this._poolStrength));
206       }
207       this._fireEvent("progress", this.getProgress());
208     }
209   },
210   
211   /** Is the generator ready? */
212   isReady: function (paranoia) {
213     var entropyRequired = this._PARANOIA_LEVELS[ (paranoia !== undefined) ? paranoia : this._defaultParanoia ];
214   
215     if (this._strength && this._strength >= entropyRequired) {
216       return (this._poolEntropy[0] > this._BITS_PER_RESEED && (new Date()).valueOf() > this._nextReseed) ?
217         this._REQUIRES_RESEED | this._READY :
218         this._READY;
219     } else {
220       return (this._poolStrength >= entropyRequired) ?
221         this._REQUIRES_RESEED | this._NOT_READY :
222         this._NOT_READY;
223     }
224   },
225   
226   /** Get the generator's progress toward readiness, as a fraction */
227   getProgress: function (paranoia) {
228     var entropyRequired = this._PARANOIA_LEVELS[ paranoia ? paranoia : this._defaultParanoia ];
229   
230     if (this._strength >= entropyRequired) {
231       return 1.0;
232     } else {
233       return (this._poolStrength > entropyRequired) ?
234         1.0 :
235         this._poolStrength / entropyRequired;
236     }
237   },
238   
239   /** start the built-in entropy collectors */
240   startCollectors: function () {
241     if (this._collectorsStarted) { return; }
242   
243     this._eventListener = {
244       loadTimeCollector: this._bind(this._loadTimeCollector),
245       mouseCollector: this._bind(this._mouseCollector),
246       keyboardCollector: this._bind(this._keyboardCollector),
247       accelerometerCollector: this._bind(this._accelerometerCollector)
248     }
249 
250     if (window.addEventListener) {
251       window.addEventListener("load", this._eventListener.loadTimeCollector, false);
252       window.addEventListener("mousemove", this._eventListener.mouseCollector, false);
253       window.addEventListener("keypress", this._eventListener.keyboardCollector, false);
254       window.addEventListener("devicemotion", this._eventListener.accelerometerCollector, false);
255     } else if (document.attachEvent) {
256       document.attachEvent("onload", this._eventListener.loadTimeCollector);
257       document.attachEvent("onmousemove", this._eventListener.mouseCollector);
258       document.attachEvent("keypress", this._eventListener.keyboardCollector);
259     } else {
260       throw new sjcl.exception.bug("can't attach event");
261     }
262   
263     this._collectorsStarted = true;
264   },
265   
266   /** stop the built-in entropy collectors */
267   stopCollectors: function () {
268     if (!this._collectorsStarted) { return; }
269   
270     if (window.removeEventListener) {
271       window.removeEventListener("load", this._eventListener.loadTimeCollector, false);
272       window.removeEventListener("mousemove", this._eventListener.mouseCollector, false);
273       window.removeEventListener("keypress", this._eventListener.keyboardCollector, false);
274       window.removeEventListener("devicemotion", this._eventListener.accelerometerCollector, false);
275     } else if (document.detachEvent) {
276       document.detachEvent("onload", this._eventListener.loadTimeCollector);
277       document.detachEvent("onmousemove", this._eventListener.mouseCollector);
278       document.detachEvent("keypress", this._eventListener.keyboardCollector);
279     }
280 
281     this._collectorsStarted = false;
282   },
283   
284   /* use a cookie to store entropy.
285   useCookie: function (all_cookies) {
286       throw new sjcl.exception.bug("random: useCookie is unimplemented");
287   },*/
288   
289   /** add an event listener for progress or seeded-ness. */
290   addEventListener: function (name, callback) {
291     this._callbacks[name][this._callbackI++] = callback;
292   },
293   
294   /** remove an event listener for progress or seeded-ness */
295   removeEventListener: function (name, cb) {
296     var i, j, cbs=this._callbacks[name], jsTemp=[];
297 
298     /* I'm not sure if this is necessary; in C++, iterating over a
299      * collection and modifying it at the same time is a no-no.
300      */
301 
302     for (j in cbs) {
303       if (cbs.hasOwnProperty(j) && cbs[j] === cb) {
304         jsTemp.push(j);
305       }
306     }
307 
308     for (i=0; i<jsTemp.length; i++) {
309       j = jsTemp[i];
310       delete cbs[j];
311     }
312   },
313   
314   _bind: function (func) {
315     var that = this;
316     return function () {
317       func.apply(that, arguments);
318     };
319   },
320 
321   /** Generate 4 random words, no reseed, no gate.
322    * @private
323    */
324   _gen4words: function () {
325     for (var i=0; i<4; i++) {
326       this._counter[i] = this._counter[i]+1 | 0;
327       if (this._counter[i]) { break; }
328     }
329     return this._cipher.encrypt(this._counter);
330   },
331   
332   /* Rekey the AES instance with itself after a request, or every _MAX_WORDS_PER_BURST words.
333    * @private
334    */
335   _gate: function () {
336     this._key = this._gen4words().concat(this._gen4words());
337     this._cipher = new sjcl.cipher.aes(this._key);
338   },
339   
340   /** Reseed the generator with the given words
341    * @private
342    */
343   _reseed: function (seedWords) {
344     this._key = sjcl.hash.sha256.hash(this._key.concat(seedWords));
345     this._cipher = new sjcl.cipher.aes(this._key);
346     for (var i=0; i<4; i++) {
347       this._counter[i] = this._counter[i]+1 | 0;
348       if (this._counter[i]) { break; }
349     }
350   },
351   
352   /** reseed the data from the entropy pools
353    * @param full If set, use all the entropy pools in the reseed.
354    */
355   _reseedFromPools: function (full) {
356     var reseedData = [], strength = 0, i;
357   
358     this._nextReseed = reseedData[0] =
359       (new Date()).valueOf() + this._MILLISECONDS_PER_RESEED;
360     
361     for (i=0; i<16; i++) {
362       /* On some browsers, this is cryptographically random.  So we might
363        * as well toss it in the pot and stir...
364        */
365       reseedData.push(Math.random()*0x100000000|0);
366     }
367     
368     for (i=0; i<this._pools.length; i++) {
369      reseedData = reseedData.concat(this._pools[i].finalize());
370      strength += this._poolEntropy[i];
371      this._poolEntropy[i] = 0;
372    
373      if (!full && (this._reseedCount & (1<<i))) { break; }
374     }
375   
376     /* if we used the last pool, push a new one onto the stack */
377     if (this._reseedCount >= 1 << this._pools.length) {
378      this._pools.push(new sjcl.hash.sha256());
379      this._poolEntropy.push(0);
380     }
381   
382     /* how strong was this reseed? */
383     this._poolStrength -= strength;
384     if (strength > this._strength) {
385       this._strength = strength;
386     }
387   
388     this._reseedCount ++;
389     this._reseed(reseedData);
390   },
391   
392   _keyboardCollector: function () {
393     this._addCurrentTimeToEntropy(1);
394   },
395   
396   _mouseCollector: function (ev) {
397     var x = ev.x || ev.clientX || ev.offsetX || 0, y = ev.y || ev.clientY || ev.offsetY || 0;
398     sjcl.random.addEntropy([x,y], 2, "mouse");
399     this._addCurrentTimeToEntropy(0);
400   },
401   
402   _loadTimeCollector: function () {
403     this._addCurrentTimeToEntropy(2);
404   },
405 
406   _addCurrentTimeToEntropy: function (estimatedEntropy) {
407     if (window && window.performance && typeof window.performance.now === "function") {
408       //how much entropy do we want to add here?
409       sjcl.random.addEntropy(window.performance.now(), estimatedEntropy, "loadtime");
410     } else {
411       sjcl.random.addEntropy((new Date()).valueOf(), estimatedEntropy, "loadtime");
412     }
413   },
414   _accelerometerCollector: function (ev) {
415     var ac = ev.accelerationIncludingGravity.x||ev.accelerationIncludingGravity.y||ev.accelerationIncludingGravity.z;
416     if(window.orientation){
417       var or = window.orientation;
418       if (typeof or === "number") {
419         sjcl.random.addEntropy(or, 1, "accelerometer");
420       }
421     }
422     if (ac) {
423       sjcl.random.addEntropy(ac, 2, "accelerometer");
424     }
425     this._addCurrentTimeToEntropy(0);
426   },
427 
428   _fireEvent: function (name, arg) {
429     var j, cbs=sjcl.random._callbacks[name], cbsTemp=[];
430     /* TODO: there is a race condition between removing collectors and firing them */
431 
432     /* I'm not sure if this is necessary; in C++, iterating over a
433      * collection and modifying it at the same time is a no-no.
434      */
435 
436     for (j in cbs) {
437       if (cbs.hasOwnProperty(j)) {
438         cbsTemp.push(cbs[j]);
439       }
440     }
441 
442     for (j=0; j<cbsTemp.length; j++) {
443       cbsTemp[j](arg);
444     }
445   }
446 };

448 /** an instance for the prng.
449 * @see sjcl.prng
450 */
451 sjcl.random = new sjcl.prng(6);

453 (function(){
454   try {
455     var buf, crypt, getRandomValues, ab;
456     // get cryptographically strong entropy depending on runtime environment
457     if (typeof module !== 'undefined' && module.exports && (crypt = require('crypto')) && crypt.randomBytes) {
458       buf = crypt.randomBytes(1024/8);
459       buf = new Uint32Array(new Uint8Array(buf).buffer);
460       sjcl.random.addEntropy(buf, 1024, "crypto.randomBytes");
461 
462     } else if (window && Uint32Array) {
463       ab = new Uint32Array(32);
464       if (window.crypto && window.crypto.getRandomValues) {
465         window.crypto.getRandomValues(ab);
466       } else if (window.msCrypto && window.msCrypto.getRandomValues) {
467         window.msCrypto.getRandomValues(ab);
468       } else {
469         return;
470       }
471 
472       // get cryptographically strong entropy in Webkit
473       sjcl.random.addEntropy(ab, 1024, "crypto.getRandomValues");
474 
475     } else {
476       // no getRandomValues :-(
477     }
478   } catch (e) {
479     console.log("There was an error collecting entropy from the browser:");
480     console.log(e);
481     //we do not want the library to fail due to randomness not being maintained.
482   }
483 }());
