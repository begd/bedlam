  8 /** @namespace UTF-8 strings */
  9 sjcl.codec.utf8String = {
 10   /** Convert from a bitArray to a UTF-8 string. */
 11   fromBits: function (arr) {
 12     var out = "", bl = sjcl.bitArray.bitLength(arr), i, tmp;
 13     for (i=0; i<bl/8; i++) {
 14       if ((i&3) === 0) {
 15         tmp = arr[i/4];
 16       }
 17       out += String.fromCharCode(tmp >>> 24);
 18       tmp <<= 8;
 19     }
 20     return decodeURIComponent(escape(out));
 21   },
 22   
 23   /** Convert from a UTF-8 string to a bitArray. */
 24   toBits: function (str) {
 25     str = unescape(encodeURIComponent(str));
 26     var out = [], i, tmp=0;
 27     for (i=0; i<str.length; i++) {
 28       tmp = tmp << 8 | str.charCodeAt(i);
 29       if ((i&3) === 3) {
 30         out.push(tmp);
 31         tmp = 0;
 32       }
 33     }
 34     if (i&3) {
 35       out.push(sjcl.bitArray.partial(8*(i&3), tmp));
 36     }
 37     return out;
 38   }
 39 };
