  8  /** @namespace JSON encapsulation */
  9  sjcl.json = {
 10   /** Default values for encryption */
 11   defaults: { v:1, iter:1000, ks:128, ts:64, mode:"ccm", adata:"", cipher:"aes" },

 13   /** Simple encryption function.
 14    * @param {String|bitArray} password The password or key.
 15    * @param {String} plaintext The data to encrypt.
 16    * @param {Object} [params] The parameters including tag, iv and salt.
 17    * @param {Object} [rp] A returned version with filled-in parameters.
 18    * @return {Object} The cipher raw data.
 19    * @throws {sjcl.exception.invalid} if a parameter is invalid.
 20    */
 21   _encrypt: function (password, plaintext, params, rp) {
 22     params = params || {};
 23     rp = rp || {};
 24 
 25     var j = sjcl.json, p = j._add({ iv: sjcl.random.randomWords(4,0) },
 26                                   j.defaults), tmp, prp, adata;
 27     j._add(p, params);
 28     adata = p.adata;
 29     if (typeof p.salt === "string") {
 30       p.salt = sjcl.codec.base64.toBits(p.salt);
 31     }
 32     if (typeof p.iv === "string") {
 33       p.iv = sjcl.codec.base64.toBits(p.iv);
 34     }
 35 
 36     if (!sjcl.mode[p.mode] ||
 37         !sjcl.cipher[p.cipher] ||
 38         (typeof password === "string" && p.iter <= 100) ||
 39         (p.ts !== 64 && p.ts !== 96 && p.ts !== 128) ||
 40         (p.ks !== 128 && p.ks !== 192 && p.ks !== 256) ||
 41         (p.iv.length < 2 || p.iv.length > 4)) {
 42       throw new sjcl.exception.invalid("json encrypt: invalid parameters");
 43     }
 44 
 45     if (typeof password === "string") {
 46       tmp = sjcl.misc.cachedPbkdf2(password, p);
 47       password = tmp.key.slice(0,p.ks/32);
 48       p.salt = tmp.salt;
 49     } else if (sjcl.ecc && password instanceof sjcl.ecc.elGamal.publicKey) {
 50       tmp = password.kem();
 51       p.kemtag = tmp.tag;
 52       password = tmp.key.slice(0,p.ks/32);
 53     }
 54     if (typeof plaintext === "string") {
 55       plaintext = sjcl.codec.utf8String.toBits(plaintext);
 56     }
 57     if (typeof adata === "string") {
 58       adata = sjcl.codec.utf8String.toBits(adata);
 59     }
 60     prp = new sjcl.cipher[p.cipher](password);
 61 
 62     /* return the json data */
 63     j._add(rp, p);
 64     rp.key = password;
 65 
 66     /* do the encryption */
 67     p.ct = sjcl.mode[p.mode].encrypt(prp, plaintext, p.iv, adata, p.ts);
 68 
 69     //return j.encode(j._subtract(p, j.defaults));
 70     return p;
 71   },
 72 
 73   /** Simple encryption function.
 74    * @param {String|bitArray} password The password or key.
 75    * @param {String} plaintext The data to encrypt.
 76    * @param {Object} [params] The parameters including tag, iv and salt.
 77    * @param {Object} [rp] A returned version with filled-in parameters.
 78    * @return {String} The ciphertext serialized data.
 79    * @throws {sjcl.exception.invalid} if a parameter is invalid.
 80    */
 81   encrypt: function (password, plaintext, params, rp) {
 82     var j = sjcl.json, p = j._encrypt.apply(j, arguments);
 83     return j.encode(p);
 84   },
 85 
 86   /** Simple decryption function.
 87    * @param {String|bitArray} password The password or key.
 88    * @param {Object} ciphertext The cipher raw data to decrypt.
 89    * @param {Object} [params] Additional non-default parameters.
 90    * @param {Object} [rp] A returned object with filled parameters.
 91    * @return {String} The plaintext.
 92    * @throws {sjcl.exception.invalid} if a parameter is invalid.
 93    * @throws {sjcl.exception.corrupt} if the ciphertext is corrupt.
 94    */
 95   _decrypt: function (password, ciphertext, params, rp) {
 96     params = params || {};
 97     rp = rp || {};
 98 
 99     var j = sjcl.json, p = j._add(j._add(j._add({},j.defaults),ciphertext), params, true), ct, tmp, prp, adata=p.adata;
100     if (typeof p.salt === "string") {
101       p.salt = sjcl.codec.base64.toBits(p.salt);
102     }
103     if (typeof p.iv === "string") {
104       p.iv = sjcl.codec.base64.toBits(p.iv);
105     }
106 
107     if (!sjcl.mode[p.mode] ||
108         !sjcl.cipher[p.cipher] ||
109         (typeof password === "string" && p.iter <= 100) ||
110         (p.ts !== 64 && p.ts !== 96 && p.ts !== 128) ||
111         (p.ks !== 128 && p.ks !== 192 && p.ks !== 256) ||
112         (!p.iv) ||
113         (p.iv.length < 2 || p.iv.length > 4)) {
114       throw new sjcl.exception.invalid("json decrypt: invalid parameters");
115     }
116 
117     if (typeof password === "string") {
118       tmp = sjcl.misc.cachedPbkdf2(password, p);
119       password = tmp.key.slice(0,p.ks/32);
120       p.salt  = tmp.salt;
121     } else if (sjcl.ecc && password instanceof sjcl.ecc.elGamal.secretKey) {
122       password = password.unkem(sjcl.codec.base64.toBits(p.kemtag)).slice(0,p.ks/32);
123     }
124     if (typeof adata === "string") {
125       adata = sjcl.codec.utf8String.toBits(adata);
126     }
127     prp = new sjcl.cipher[p.cipher](password);
128 
129     /* do the decryption */
130     ct = sjcl.mode[p.mode].decrypt(prp, p.ct, p.iv, adata, p.ts);
131 
132     /* return the json data */
133     j._add(rp, p);
134     rp.key = password;
135 
136     return sjcl.codec.utf8String.fromBits(ct);
137   },

139   /** Simple decryption function.
140    * @param {String|bitArray} password The password or key.
141    * @param {String} ciphertext The ciphertext to decrypt.
142    * @param {Object} [params] Additional non-default parameters.
143    * @param {Object} [rp] A returned object with filled parameters.
144    * @return {String} The plaintext.
145    * @throws {sjcl.exception.invalid} if a parameter is invalid.
146    * @throws {sjcl.exception.corrupt} if the ciphertext is corrupt.
147    */
148   decrypt: function (password, ciphertext, params, rp) {
149     var j = sjcl.json;
150     return j._decrypt(password, j.decode(ciphertext), params, rp);
151   },

153   /** Encode a flat structure into a JSON string.
154    * @param {Object} obj The structure to encode.
155    * @return {String} A JSON string.
156    * @throws {sjcl.exception.invalid} if obj has a non-alphanumeric property.
157    * @throws {sjcl.exception.bug} if a parameter has an unsupported type.
158    */
159   encode: function (obj) {
160     var i, out='{', comma='';
161     for (i in obj) {
162       if (obj.hasOwnProperty(i)) {
163         if (!i.match(/^[a-z0-9]+$/i)) {
164           throw new sjcl.exception.invalid("json encode: invalid property name");
165         }
166         out += comma + '"' + i + '":';
167         comma = ',';
168 
169         switch (typeof obj[i]) {
170           case 'number':
171           case 'boolean':
172             out += obj[i];
173             break;
174 
175           case 'string':
176             out += '"' + escape(obj[i]) + '"';
177             break;
178 
179           case 'object':
180             out += '"' + sjcl.codec.base64.fromBits(obj[i],0) + '"';
181             break;
182 
183           default:
184             throw new sjcl.exception.bug("json encode: unsupported type");
185         }
186       }
187     }
188     return out+'}';
189   },
190   
191   /** Decode a simple (flat) JSON string into a structure.  The ciphertext,
192    * adata, salt and iv will be base64-decoded.
193    * @param {String} str The string.
194    * @return {Object} The decoded structure.
195    * @throws {sjcl.exception.invalid} if str isn't (simple) JSON.
196    */
197   decode: function (str) {
198     str = str.replace(/\s/g,'');
199     if (!str.match(/^\{.*\}$/)) { 
200       throw new sjcl.exception.invalid("json decode: this isn't json!");
201     }
202     var a = str.replace(/^\{|\}$/g, '').split(/,/), out={}, i, m;
203     for (i=0; i<a.length; i++) {
204       if (!(m=a[i].match(/^(?:(["']?)([a-z][a-z0-9]*)\1):(?:(\d+)|"([a-z0-9+\/%*_.@=\-]*)")$/i))) {
205         throw new sjcl.exception.invalid("json decode: this isn't json!");
206       }
207       if (m[3]) {
208         out[m[2]] = parseInt(m[3],10);
209       } else {
210         out[m[2]] = m[2].match(/^(ct|salt|iv)$/) ? sjcl.codec.base64.toBits(m[4]) : unescape(m[4]);
211       }
212     }
213     return out;
214   },

216   /** Insert all elements of src into target, modifying and returning target.
217    * @param {Object} target The object to be modified.
218    * @param {Object} src The object to pull data from.
219    * @param {boolean} [requireSame=false] If true, throw an exception if any field of target differs from corresponding field of src.
220    * @return {Object} target.
221    * @private
222    */
223   _add: function (target, src, requireSame) {
224     if (target === undefined) { target = {}; }
225     if (src === undefined) { return target; }
226     var i;
227     for (i in src) {
228       if (src.hasOwnProperty(i)) {
229         if (requireSame && target[i] !== undefined && target[i] !== src[i]) {
230           throw new sjcl.exception.invalid("required parameter overridden");
231         }
232         target[i] = src[i];
233       }
234     }
235     return target;
236   },
237   
238   /** Remove all elements of minus from plus.  Does not modify plus.
239    * @private
240    */
241   _subtract: function (plus, minus) {
242     var out = {}, i;
243 
244     for (i in plus) {
245       if (plus.hasOwnProperty(i) && plus[i] !== minus[i]) {
246         out[i] = plus[i];
247       }
248     }
249 
250     return out;
251   },
252   
253   /** Return only the specified elements of src.
254    * @private
255    */
256   _filter: function (src, filter) {
257     var out = {}, i;
258     for (i=0; i<filter.length; i++) {
259       if (src[filter[i]] !== undefined) {
260         out[filter[i]] = src[filter[i]];
261       }
262     }
263     return out;
264   }

290 /** Cached PBKDF2 key derivation.
291  * @param {String} password The password.
292  * @param {Object} [obj] The derivation params (iteration count and optional salt).
293  * @return {Object} The derived data in key, the salt in salt.
294  */
295 sjcl.misc.cachedPbkdf2 = function (password, obj) {
296   var cache = sjcl.misc._pbkdf2Cache, c, cp, str, salt, iter;
297   
298   obj = obj || {};
299   iter = obj.iter || 1000;
300   
301   /* open the cache for this password and iteration count */
302   cp = cache[password] = cache[password] || {};
303   c = cp[iter] = cp[iter] || { firstSalt: (obj.salt && obj.salt.length) ?
304                      obj.salt.slice(0) : sjcl.random.randomWords(2,0) };
305           
306   salt = (obj.salt === undefined) ? c.firstSalt : obj.salt;
307   
308   c[salt] = c[salt] || sjcl.misc.pbkdf2(password, salt, obj.iter);
309   return { key: c[salt].slice(0), salt:salt.slice(0) };
310 };
