  8 /** @namespace Hexadecimal */
  9 sjcl.codec.hex = {
 10   /** Convert from a bitArray to a hex string. */
 11   fromBits: function (arr) {
 12     var out = "", i, x;
 13     for (i=0; i<arr.length; i++) {
 14       out += ((arr[i]|0)+0xF00000000000).toString(16).substr(4);
 15     }
 16     return out.substr(0, sjcl.bitArray.bitLength(arr)/4);//.replace(/(.{8})/g, "$1 ");
 17   },
 18   /** Convert from a hex string to a bitArray. */
 19   toBits: function (str) {
 20     var i, out=[], len;
 21     str = str.replace(/\s|0x/g, "");
 22     len = str.length;
 23     str = str + "00000000";
 24     for (i=0; i<str.length; i+=8) {
 25       out.push(parseInt(str.substr(i,8),16)^0);
 26     }
 27     return sjcl.bitArray.clamp(out, len*4);
 28   }
 29 };
