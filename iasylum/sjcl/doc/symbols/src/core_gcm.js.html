  6 /** @namespace Galois/Counter mode. */
  7 sjcl.mode.gcm = {
  8   /** The name of the mode.
  9    * @constant
 10    */
 11   name: "gcm",
 12   
 13   /** Encrypt in GCM mode.
 14    * @static
 15    * @param {Object} prf The pseudorandom function.  It must have a block size of 16 bytes.
 16    * @param {bitArray} plaintext The plaintext data.
 17    * @param {bitArray} iv The initialization value.
 18    * @param {bitArray} [adata=[]] The authenticated data.
 19    * @param {Number} [tlen=128] The desired tag length, in bits.
 20    * @return {bitArray} The encrypted data, an array of bytes.
 21    */
 22   encrypt: function (prf, plaintext, iv, adata, tlen) {
 23     var out, data = plaintext.slice(0), w=sjcl.bitArray;
 24     tlen = tlen || 128;
 25     adata = adata || [];
 26 
 27     // encrypt and tag
 28     out = sjcl.mode.gcm._ctrMode(true, prf, data, adata, iv, tlen);
 29 
 30     return w.concat(out.data, out.tag);
 31   },
 32   
 33   /** Decrypt in GCM mode.
 34    * @static
 35    * @param {Object} prf The pseudorandom function.  It must have a block size of 16 bytes.
 36    * @param {bitArray} ciphertext The ciphertext data.
 37    * @param {bitArray} iv The initialization value.
 38    * @param {bitArray} [adata=[]] The authenticated data.
 39    * @param {Number} [tlen=128] The desired tag length, in bits.
 40    * @return {bitArray} The decrypted data.
 41    */
 42   decrypt: function (prf, ciphertext, iv, adata, tlen) {
 43     var out, data = ciphertext.slice(0), tag, w=sjcl.bitArray, l=w.bitLength(data);
 44     tlen = tlen || 128;
 45     adata = adata || [];
 46 
 47     // Slice tag out of data
 48     if (tlen <= l) {
 49       tag = w.bitSlice(data, l-tlen);
 50       data = w.bitSlice(data, 0, l-tlen);
 51     } else {
 52       tag = data;
 53       data = [];
 54     }
 55 
 56     // decrypt and tag
 57     out = sjcl.mode.gcm._ctrMode(false, prf, data, adata, iv, tlen);
 58 
 59     if (!w.equal(out.tag, tag)) {
 60       throw new sjcl.exception.corrupt("gcm: tag doesn't match");
 61     }
 62     return out.data;
 63   },
 64 
 65   /* Compute the galois multiplication of X and Y
 66    * @private
 67    */
 68   _galoisMultiply: function (x, y) {
 69     var i, j, xi, Zi, Vi, lsb_Vi, w=sjcl.bitArray, xor=w._xor4;
 70 
 71     Zi = [0,0,0,0];
 72     Vi = y.slice(0);
 73 
 74     // Block size is 128 bits, run 128 times to get Z_128
 75     for (i=0; i<128; i++) {
 76       xi = (x[Math.floor(i/32)] & (1 << (31-i%32))) !== 0;
 77       if (xi) {
 78         // Z_i+1 = Z_i ^ V_i
 79         Zi = xor(Zi, Vi);
 80       }
 81 
 82       // Store the value of LSB(V_i)
 83       lsb_Vi = (Vi[3] & 1) !== 0;
 84 
 85       // V_i+1 = V_i >> 1
 86       for (j=3; j>0; j--) {
 87         Vi[j] = (Vi[j] >>> 1) | ((Vi[j-1]&1) << 31);
 88       }
 89       Vi[0] = Vi[0] >>> 1;
 90 
 91       // If LSB(V_i) is 1, V_i+1 = (V_i >> 1) ^ R
 92       if (lsb_Vi) {
 93         Vi[0] = Vi[0] ^ (0xe1 << 24);
 94       }
 95     }
 96     return Zi;
 97   },
 98 
 99   _ghash: function(H, Y0, data) {
100     var Yi, i, l = data.length;
101 
102     Yi = Y0.slice(0);
103     for (i=0; i<l; i+=4) {
104       Yi[0] ^= 0xffffffff&data[i];
105       Yi[1] ^= 0xffffffff&data[i+1];
106       Yi[2] ^= 0xffffffff&data[i+2];
107       Yi[3] ^= 0xffffffff&data[i+3];
108       Yi = sjcl.mode.gcm._galoisMultiply(Yi, H);
109     }
110     return Yi;
111   },
112 
113   /** GCM CTR mode.
114    * Encrypt or decrypt data and tag with the prf in GCM-style CTR mode.
115    * @param {Boolean} encrypt True if encrypt, false if decrypt.
116    * @param {Object} prf The PRF.
117    * @param {bitArray} data The data to be encrypted or decrypted.
118    * @param {bitArray} iv The initialization vector.
119    * @param {bitArray} adata The associated data to be tagged.
120    * @param {Number} tlen The length of the tag, in bits.
121    */
122   _ctrMode: function(encrypt, prf, data, adata, iv, tlen) {
123     var H, J0, S0, enc, i, ctr, tag, last, l, bl, abl, ivbl, w=sjcl.bitArray, xor=w._xor4;
124 
125     // Calculate data lengths
126     l = data.length;
127     bl = w.bitLength(data);
128     abl = w.bitLength(adata);
129     ivbl = w.bitLength(iv);
130 
131     // Calculate the parameters
132     H = prf.encrypt([0,0,0,0]);
133     if (ivbl === 96) {
134       J0 = iv.slice(0);
135       J0 = w.concat(J0, [1]);
136     } else {
137       J0 = sjcl.mode.gcm._ghash(H, [0,0,0,0], iv);
138       J0 = sjcl.mode.gcm._ghash(H, J0, [0,0,Math.floor(ivbl/0x100000000),ivbl&0xffffffff]);
139     }
140     S0 = sjcl.mode.gcm._ghash(H, [0,0,0,0], adata);
141 
142     // Initialize ctr and tag
143     ctr = J0.slice(0);
144     tag = S0.slice(0);
145 
146     // If decrypting, calculate hash
147     if (!encrypt) {
148       tag = sjcl.mode.gcm._ghash(H, S0, data);
149     }
150 
151     // Encrypt all the data
152     for (i=0; i<l; i+=4) {
153        ctr[3]++;
154        enc = prf.encrypt(ctr);
155        data[i]   ^= enc[0];
156        data[i+1] ^= enc[1];
157        data[i+2] ^= enc[2];
158        data[i+3] ^= enc[3];
159     }
160     data = w.clamp(data, bl);
161 
162     // If encrypting, calculate hash
163     if (encrypt) {
164       tag = sjcl.mode.gcm._ghash(H, S0, data);
165     }
166 
167     // Calculate last block from bit lengths, ugly because bitwise operations are 32-bit
168     last = [
169       Math.floor(abl/0x100000000), abl&0xffffffff,
170       Math.floor(bl/0x100000000), bl&0xffffffff
171     ];
172 
173     // Calculate the final tag block
174     tag = sjcl.mode.gcm._ghash(H, tag, last);
175     enc = prf.encrypt(J0);
176     tag[0] ^= enc[0];
177     tag[1] ^= enc[1];
178     tag[2] ^= enc[2];
179     tag[3] ^= enc[3];
180 
181     return { tag:w.bitSlice(tag, 0, tlen), data:data };
182   }
183 };
